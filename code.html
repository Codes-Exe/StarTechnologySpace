Star Teknoloji JavaScript

Bootstrap your JavaScript projects with our powerful tools and frameworks. Build faster, scale better, and launch sooner.

View on GitHub 
import { StarTech } from '@starteknoloji/core';

// Initialize the Star Teknoloji framework
const app = new StarTech({
  mode: 'production',
  plugins: ['analytics', 'performance']
});

// Create a new component
app.createComponent('StarButton', {
  template: '<button class="star-btn">{{ label }}</button>',
  props: {
    label: 'Click me!'
  },
  methods: {
    onClick() {
      this.emit('star-clicked');
    }
  }
});

// Launch your application
app.launch();

export default {
  props: {
    title: String,
    description: String,
    buttonText: {
      type: String,
      default: 'Click me'
    }
  },
  methods: {
    onAction() {
      this.$emit('action-clicked');
      console.log('Button clicked!');
    }
  }
}
